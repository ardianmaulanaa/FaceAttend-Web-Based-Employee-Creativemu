longitude: number },  pointB: { latitude: number; longitude: number },) {  const earthRadius = 6371000;  const dLat = toRadians(pointB.latitude - pointA.latitude);  const dLon = toRadians(pointB.longitude - pointA.longitude);  const lat1 = toRadians(pointA.latitude);  const lat2 = toRadians(pointB.latitude);  const a =    Math.sin(dLat / 2) * Math.sin(dLat / 2) +    Math.cos(lat1) * Math.cos(lat2) * Math.sin(dLon / 2) * Math.sin(dLon / 2);  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));  return earthRadius * c;}function resolveMainLocation(latitude: number, longitude: number) {  const nearest = MAIN_LOCATIONS.map((location) => ({    ...location,    distance: distanceMeters(      { latitude, longitude },      { latitude: location.latitude, longitude: location.longitude },    ),  })).sort((a, b) => a.distance - b.distance)[0];
